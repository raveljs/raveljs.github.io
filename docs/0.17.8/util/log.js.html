An abstraction for a logging service which behaves like intel

Source 
class Logger {
  constructor(source) {
    intel.basicConfig({
      'format': '[%(date)s] %(name)s.%(levelname)s: %(message)s'
    });

    this[sLogger] = intel.getLogger(source);

    this.TRACE = 'TRACE';
    this.VERBOSE = 'VERBOSE';
    this.DEBUG = 'DEBUG';
    this.INFO = 'INFO';
    this.WARN = 'WARN';
    this.ERROR = 'ERROR';
    this.CRITICAL = 'CRITICAL';
    this.NONE = 'NONE';
    this.ALL = 'ALL';
  }
